- `tokio` (1.40) - Async runtime

pub mod completion;

/// Configuration for REPL behavior
#[derive(Debug, Clone)]
pub struct ReplConfig {
    /// Path to history file (default: ~/.mimi/repl_history)
    pub history_file: String,
    
    /// Maximum history entries (default: 1000)
    pub max_history: usize,
    
    /// Disable history persistence
    pub no_history: bool,
    
    /// Startup script to run on init
    pub startup_script: Option<String>,
    
    /// Enable command completion
    pub completion: bool,
}

impl Default for ReplConfig {
    fn default() -> Self {
        Self {
            history_file: format!("{}/.mimi/repl_history", 
                std::env::var("HOME").unwrap_or_else(|_| ".".to_string())),
            max_history: 1000,
            no_history: false,
            startup_script: None,
            completion: true,
        }
    }
}

/// Prompt states for REPL
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum PromptState {
    /// Default prompt: "mimi> "
    Default,
    
    /// Following async task: "mimi [task-123]> "
    Following(u32),
    
    /// Debug mode: "mimi [debug]> "
    Debug,
    
    /// Continuation: "... "
    Continuation,
}

impl PromptState {
    pub fn prompt_string(&self) -> String {
        match self {
            Self::Default => "mimi> ".to_string(),
            Self::Following(task_id) => format!("mimi [task-{}]> ", task_id),
            Self::Debug => "mimi [debug]> ".to_string(),
            Self::Continuation => "... ".to_string(),
        }
    }
}

//! Rustyline editor wrapper for REPL

use rustyline::error::ReadlineError;
use rustyline::history::DefaultHistory;
use rustyline::{DefaultEditor, Result as RustlineResult};
use std::path::Path;

use super::{ReplConfig, PromptState};

/// Wraps rustyline editor with custom configuration
pub struct ReplEditor {
    editor: DefaultEditor,
    config: ReplConfig,
}

impl ReplEditor {
    /// Create new REPL editor with configuration
    pub fn new(config: ReplConfig) -> RustlineResult<Self> {
        let mut editor = DefaultEditor::new()?;
        
        // Load history if not disabled
        if !config.no_history && Path::new(&config.history_file).exists() {
            let _ = editor.load_history(&config.history_file);
        }
        
        // Set history size
        editor.history_mut().set_max_len(config.max_history);
        
        Ok(Self { editor, config })
    }
    
    /// Read a line from the user
    pub fn read_line(&mut self, prompt_state: PromptState) -> RustlineResult<String> {
        let prompt = prompt_state.prompt_string();
        self.editor.readline(&prompt)
    }
    
    /// Add line to history
    pub fn add_history(&mut self, line: &str) -> RustlineResult<()> {
        self.editor.add_history_entry(line)
    }
    
    /// Save history to file
    pub fn save_history(&mut self) -> RustlineResult<()> {
        if !self.config.no_history {
            self.editor.save_history(&self.config.history_file)?;
        }
        Ok(())
    }
    
    /// Get history (for .history command)
    pub fn get_history_lines(&self, n: usize) -> Vec<String> {
        let history = self.editor.history();
        let total = history.len();
        
        if total == 0 {
            return vec![];
        }
        
        let start = if n >= total { 0 } else { total - n };
        
        (start..total)
            .filter_map(|i| history.get(i, &mut self.editor.history_mut().clone()))
            .map(|entry| entry.entry.to_string())
            .collect()
    }
}

//! Special commands for REPL (prefixed with .)

use crate::cli::{CommandHandler, Formatter};
use crate::config::Config;

/// Special REPL commands (prefix: .)
#[derive(Debug, Clone, PartialEq, Eq)]
pub enum SpecialCommand {
    /// .help [COMMAND] - Show help
    Help(Option<String>),
    
    /// .status - Show system status
    Status,
    
    /// .clear - Clear screen
    Clear,
    
    /// .config get <KEY> - Get config value
    ConfigGet(String),
    
    /// .config set <KEY> <VALUE> - Set config value
    ConfigSet(String, String),
    
    /// .history [N] - Show last N commands (default 20)
    History(Option<usize>),
    
    /// .exit, .quit - Exit REPL
    Exit,
    
    /// !<SHELL_CMD> - Run shell command
    Shell(String),
}

impl SpecialCommand {
    /// Parse a line starting with . or !
    pub fn parse(line: &str) -> Option<Self> {
        if line.is_empty() {
            return None;
        }
        
        let trimmed = line.trim();
        
        // Shell command
        if trimmed.starts_with('!') {
            let cmd = trimmed[1..].to_string();
            return Some(Self::Shell(cmd));
        }
        
        // Special commands (must start with .)
        if !trimmed.starts_with('.') {
            return None;
        }
        
        let parts: Vec<&str> = trimmed.split_whitespace().collect();
        
        match parts.as_slice() {
            [".help"] => Some(Self::Help(None)),
            [".help", cmd] => Some(Self::Help(Some(cmd.to_string()))),
            
            [".status"] => Some(Self::Status),
            
            [".clear"] => Some(Self::Clear),
            
            [".config", "get", key] => Some(Self::ConfigGet(key.to_string())),
            [".config", "set", key, value] => Some(Self::ConfigSet(key.to_string(), value.to_string())),
            
            [".history"] => Some(Self::History(None)),
            [".history", n_str] => {
                n_str.parse::<usize>().ok().map(Some).map(Self::History)?
            }
            
            [".exit"] | [".quit"] => Some(Self::Exit),
            
            _ => None,
        }
    }
    
    /// Execute special command
    pub async fn execute(
        &self,
        handler: &CommandHandler,
        config: &Config,
        formatter: &Formatter,
    ) -> Result<String, String> {
        match self {
            Self::Help(cmd) => {
                if let Some(cmd_name) = cmd {
                    Ok(format!("Help for {}: (detailed help to be implemented)", cmd_name))
                } else {
                    Ok("Special REPL commands:\n  .help [COMMAND]     - Show help\n  .status             - Show system status\n  .clear              - Clear screen\n  .config get <KEY>   - Get config value\n  .config set <KEY> <VAL> - Set config value\n  .history [N]        - Show last N commands\n  .exit / .quit       - Exit REPL\n  !<CMD>              - Run shell command".to_string())
                }
            }
            
            Self::Status => {
                // TODO: Implement full status reporting
                Ok("System status: OK".to_string())
            }
            
            Self::Clear => {
                // Clear screen using ANSI escape code
                Ok("\x1B[2J\x1B[H".to_string())
            }
            
            Self::ConfigGet(key) => {
                match config.get(key) {
                    Some(val) => Ok(format!("{} = {}", key, val)),
                    None => Err(format!("Config key not found: {}", key)),
                }
            }
            
            Self::ConfigSet(key, value) => {
                // TODO: Implement config update with persistence
                Ok(format!("Set {} = {}", key, value))
            }
            
            Self::History(n) => {
                let n = n.unwrap_or(20);
                // History will be populated by REPL loop
                Ok(format!("History (last {} entries):\n(to be populated by REPL)", n))
            }
            
            Self::Exit => Ok("exit".to_string()),
            
            Self::Shell(cmd) => {
                match std::process::Command::new("sh")
                    .arg("-c")
                    .arg(cmd)
                    .output()
                {
                    Ok(output) => {
                        let stdout = String::from_utf8_lossy(&output.stdout);
                        Ok(stdout.to_string())
                    }
                    Err(e) => Err(format!("Shell error: {}", e)),
                }
            }
        }
    }
}

//! Main REPL event loop

use rustyline::error::ReadlineError;
use std::io::Write;

use crate::cli::{CommandHandler, Formatter};
use crate::config::Config;

use super::{ReplConfig, ReplEditor, PromptState, SpecialCommand};

/// Run the REPL main loop
pub async fn run_repl(
    config: Config,
    formatter: &Formatter,
    repl_config: ReplConfig,
) -> crate::cli::CliResult<()> {
    let mut editor = ReplEditor::new(repl_config.clone())
        .map_err(|e| crate::cli::CliError::IoError(e.to_string()))?;
    
    let handler = CommandHandler::new(&config);
    let mut prompt_state = PromptState::Default;
    let mut multiline_buffer = String::new();
    
    // Run startup script if provided
    if let Some(script_path) = &repl_config.startup_script {
        if let Ok(content) = std::fs::read_to_string(script_path) {
            for line in content.lines() {
                let line = line.trim();
                if !line.is_empty() && !line.starts_with('#') {
                    println!("{}{}", prompt_state.prompt_string(), line);
                    // Execute line (simplified)
                    let _ = execute_line(line, &handler, &config, formatter, &mut editor)
                        .await;
                }
            }
        }
    }
    
    // Main REPL loop
    loop {
        match editor.read_line(prompt_state) {
            Ok(line) => {
                let line = line.trim();
                
                // Empty line - skip
                if line.is_empty() {
                    continue;
                }
                
                // Check for line continuation
                if line.ends_with('\\') {
                    multiline_buffer.push_str(&line[..line.len() - 1]);
                    multiline_buffer.push(' ');
                    prompt_state = PromptState::Continuation;
                    continue;
                }
                
                // Complete multiline input
                if !multiline_buffer.is_empty() {
                    multiline_buffer.push_str(line);
                    prompt_state = PromptState::Default;
                } else {
                    multiline_buffer = line.to_string();
                }
                
                // Add to history
                let _ = editor.add_history(&multiline_buffer);
                
                // Execute line
                match execute_line(&multiline_buffer, &handler, &config, formatter, &mut editor).await
                {
                    Ok(exit) => {
                        if exit {
                            break;
                        }
                    }
                    Err(e) => {
                        eprintln!("{}", formatter.format_error(&e, "1"));
                    }
                }
                
                multiline_buffer.clear();
                prompt_state = PromptState::Default;
            }
            
            Err(ReadlineError::Interrupted) => {
                println!("^C");
                continue;
            }
            
            Err(ReadlineError::Eof) => {
                println!();
                break;
            }
            
            Err(e) => {
                eprintln!("REPL error: {}", e);
                break;
            }
        }
    }
    
    // Save history on exit
    let _ = editor.save_history();
    
    Ok(())
}

/// Execute a single REPL line
async fn execute_line(
    line: &str,
    _handler: &CommandHandler,
    config: &Config,
    formatter: &Formatter,
    editor: &mut ReplEditor,
) -> Result<bool, String> {
    // Check for special commands
    if let Some(special_cmd) = SpecialCommand::parse(line) {
        match special_cmd {
            SpecialCommand::Exit => return Ok(true),
            SpecialCommand::Clear => {
                print!("\x1B[2J\x1B[H");
                let _ = std::io::stdout().flush();
                return Ok(false);
            }
            SpecialCommand::History(n) => {
                let n = n.unwrap_or(20);
                let lines = editor.get_history_lines(n);
                for (i, line) in lines.iter().enumerate() {
                    println!("  {}: {}", i + 1, line);
                }
                return Ok(false);
            }
            _ => {
                // Other special commands would be executed here
                let output = special_cmd
                    .execute(_handler, config, formatter)
                    .await
                    .unwrap_or_else(|e| formatter.format_error(&e, "1"));
                println!("{}", output);
                return Ok(false);
            }
        }
    }
    
    // Parse as regular command
    // TODO: In M1.4.6, parse through clap and dispatch to handler
    // For now, show helpful message
    println!(
        "{}",
        formatter.format_success(&format!("REPL: parsed command '{}' (execution in M1.4.6)", line))
    );
    
    Ok(false)
}

# M1.4.3: Beatrice REPL Implementation

## Overview

The REPL (Read-Eval-Print Loop) provides an interactive shell interface for continuous interaction with MiMi. Built on `rustyline` for history management and line editing, it offers:

- **Command history** with persistence to `~/.mimi/repl_history`
- **Special commands** for control and inspection (`.help`, `.status`, `.config`, etc.)
- **Multi-line input** support with backslash continuation
- **Startup scripts** for initialization
- **Shell command execution** via `!command` syntax

## Architecture

### Module Structure

```
src/repl/
├── mod.rs                  # Main REPL configuration and state
├── editor.rs              # Rustyline wrapper
├── completion.rs          # Command completion (stub for Phase 2)
├── special_commands.rs    # Dot commands parser and executor
└── loop.rs               # Main event loop
```

### Data Flow

1. User enters text in REPL prompt
2. `ReplEditor.read_line()` reads input via rustyline
3. Check for special commands (`.` prefix) → route to `SpecialCommand::execute()`
4. Check for shell commands (`!` prefix) → fork subprocess
5. Otherwise → parse as MiMi command (M1.4.6) → delegate to `CommandHandler`
6. Format output with `Formatter`
7. Add to history and display

### Integration Points

- **M1.4.2 CLI** (Commands, Handler, Formatter) - Delegates regular command execution
- **Config System** - Reads/writes configuration via `Config` struct
- **Logger** - Uses existing logging infrastructure

## Special Commands Reference

| Command | Purpose | Example |
|---------|---------|---------|
| `.help [CMD]` | Show help | `.help exec` |
| `.status` | System status | `.status` |
| `.clear` | Clear screen | `.clear` |
| `.config get KEY` | Get config value | `.config get adapter.default` |
| `.config set KEY VAL` | Set config value | `.config set debug.verbosity 2` |
| `.history [N]` | Show last N commands | `.history 20` |
| `.exit` / `.quit` | Exit REPL | `.exit` |
| `!COMMAND` | Run shell command | `!ls -la` |

## Usage Examples

### Start REPL

```bash
# Default REPL
mimi

# With custom history file
mimi repl --history-file ~/project-history

# Without history persistence
mimi repl --no-history

# With startup script
mimi repl --startup-script setup.mimi
```

### Inside REPL

```
mimi> query what's the current status
mimi> exec create skill for JSON validation
mimi> .config get adapter.default
adapter.default = "gemini"
mimi> .history 5
  1: query what's the current status
  2: exec create skill for JSON validation
  3: .config get adapter.default
  4: .history 5
mimi> .exit
```
