use std::sync::atomic::{AtomicU32, Ordering};
use std::sync::Arc;

/// Tracks health status of an adapter
#[derive(Clone)]
pub struct AdapterHealth {
    pub adapter_name: String,
    pub is_healthy: bool,
    pub consecutive_failures: u32,
    pub success_count: u32,
    pub error_count: u32,
    pub last_latency_ms: u32,
    last_check_timestamp: u64, // seconds since epoch
}

impl AdapterHealth {
    /// Create new health tracker
    pub fn new(adapter_name: &str) -> Self {
        Self {
            adapter_name: adapter_name.to_string(),
            is_healthy: true,
            consecutive_failures: 0,
            success_count: 0,
            error_count: 0,
            last_latency_ms: 0,
            last_check_timestamp: current_timestamp(),
        }
    }

    /// Record successful health check
    pub fn record_success(&mut self, latency_ms: u32) {
        self.is_healthy = true;
        self.consecutive_failures = 0;
        self.success_count += 1;
        self.last_latency_ms = latency_ms;
        self.last_check_timestamp = current_timestamp();
    }

    /// Record failed health check
    pub fn record_failure(&mut self) {
        self.consecutive_failures += 1;
        self.error_count += 1;
        self.last_check_timestamp = current_timestamp();
        
        // Mark unhealthy after 3 consecutive failures
        if self.consecutive_failures >= 3 {
            self.is_healthy = false;
        }
    }

    /// Calculate success rate (0.0 to 1.0)
    pub fn success_rate(&self) -> f64 {
        let total = self.success_count as f64 + self.error_count as f64;
        if total == 0.0 {
            0.0
        } else {
            self.success_count as f64 / total
        }
    }

    /// Calculate average latency in milliseconds
    pub fn average_latency_ms(&self) -> f64 {
        if self.success_count == 0 {
            0.0
        } else {
            self.last_latency_ms as f64 // Simplified: use last latency, could be full average
        }
    }

    /// Check if adapter was checked recently (within last N seconds)
    pub fn is_recently_checked(&self, within_seconds: u64) -> bool {
        let now = current_timestamp();
        now.saturating_sub(self.last_check_timestamp) <= within_seconds
    }
}

fn current_timestamp() -> u64 {
    SystemTime::now()
        .duration_since(UNIX_EPOCH)
        .unwrap_or_default()
        .as_secs()
}

use super::{adapter::SharedAdapter, health::AdapterHealth, error::AdapterResult};
use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::RwLock;
use tokio::task::JoinHandle;
use std::time::Duration;

/// Background health checker for adapters
pub struct HealthChecker {
    check_interval_secs: u64,
}

impl HealthChecker {
    /// Create new health checker with specified interval
    pub fn new(check_interval_secs: u64) -> Self {
        Self { check_interval_secs }
    }

    /// Spawn background health check task
    pub fn spawn(
        &self,
        adapter_registry: Arc<RwLock<HashMap<String, (SharedAdapter, Arc<RwLock<AdapterHealth>>)>>>,
    ) -> JoinHandle<()> {
        let interval_secs = self.check_interval_secs;
        
        tokio::spawn(async move {
            let mut interval = tokio::time::interval(Duration::from_secs(interval_secs));
            
            loop {
                interval.tick().await;
                
                // Get adapter names to check
                let adapter_names: Vec<String> = {
                    let registry = adapter_registry.read().await;
                    registry.keys().cloned().collect()
                };

                // Check each adapter
                for name in adapter_names {
                    if let Ok(registry) = adapter_registry.read().await.get(&name).cloned() {
                        let (adapter, health) = registry;
                        
                        // Record the health check
                        // In real implementation, would send capability test
                        let latency_ms = 10; // Mock latency
                        let mut h = health.write().await;
                        h.record_success(latency_ms);
                    }
                }
            }
        })
    }
}

use serde::{Deserialize, Serialize};
use std::path::Path;
use std::fs;

/// Adapter priority configuration
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AdapterPriority {
    pub adapters: Vec<String>,
    pub check_interval_secs: u64,
}

impl AdapterPriority {
    /// Create new priority configuration
    pub fn new(adapters: Vec<String>, check_interval_secs: u64) -> Self {
        Self {
            adapters,
            check_interval_secs,
        }
    }

    /// Load priority from TOML file
    pub async fn from_file(path: &str) -> Result<Self, Box<dyn std::error::Error>> {
        let content = fs::read_to_string(path)?;
        let config: AdapterPriority = toml::from_str(&content)?;
        Ok(config)
    }

    /// Get first (primary) adapter in priority list
    pub fn first_adapter(&self) -> Option<String> {
        self.adapters.first().cloned()
    }

    /// Get fallback chain (all adapters in order)
    pub fn fallback_chain(&self) -> Vec<String> {
        self.adapters.clone()
    }

    /// Get fallback adapters (all except primary)
    pub fn fallback_adapters(&self) -> Vec<String> {
        if self.adapters.is_empty() {
            vec![]
        } else {
            self.adapters[1..].to_vec()
        }
    }
}

impl Default for AdapterPriority {
    fn default() -> Self {
        Self {
            adapters: vec!["gemini".to_string(), "ollama".to_string()],
            check_interval_secs: 30,
        }
    }
}

use super::{adapter::SharedAdapter, health::AdapterHealth, error::{AdapterResult, AdapterError}};
use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::RwLock;

/// Enhanced adapter registry with health tracking
pub struct AdapterRegistry {
    adapters: RwLock<HashMap<String, (SharedAdapter, Arc<RwLock<AdapterHealth>>)>>,
    priority: Vec<String>,
}

impl AdapterRegistry {
    /// Create new registry
    pub fn new() -> Arc<Self> {
        Arc::new(Self {
            adapters: RwLock::new(HashMap::new()),
            priority: vec!["gemini".to_string(), "ollama".to_string()],
        })
    }

    /// Create registry with custom priority
    pub fn with_priority(priority: Vec<String>) -> Arc<Self> {
        Arc::new(Self {
            adapters: RwLock::new(HashMap::new()),
            priority,
        })
    }

    /// Register adapter with health tracking
    pub async fn register_with_health(
        &self,
        name: String,
        adapter: SharedAdapter,
    ) -> AdapterResult<()> {
        let mut adapters = self.adapters.write().await;
        let health = Arc::new(RwLock::new(AdapterHealth::new(&name)));
        adapters.insert(name, (adapter, health));
        Ok(())
    }

    /// Get adapter by name (legacy, always returns adapter even if unhealthy)
    pub async fn get(&self, name: &str) -> AdapterResult<SharedAdapter> {
        let adapters = self.adapters.read().await;
        adapters
            .get(name)
            .map(|(adapter, _)| adapter.clone())
            .ok_or_else(|| AdapterError::AdapterNotFound(format!("adapter not found: {}", name)))
    }

    /// Get best available adapter (uses priority, checks health)
    pub async fn get_best_available(&self) -> AdapterResult<SharedAdapter> {
        let adapters = self.adapters.read().await;

        // Try adapters in priority order
        for adapter_name in &self.priority {
            if let Some((adapter, health)) = adapters.get(adapter_name) {
                let h = health.read().await;
                if h.is_healthy {
                    return Ok(adapter.clone());
                }
            }
        }

        // If no healthy adapter in priority list, try any healthy adapter
        for (_, (adapter, health)) in adapters.iter() {
            let h = health.read().await;
            if h.is_healthy {
                return Ok(adapter.clone());
            }
        }

        // Fallback: return first in priority order, even if unhealthy
        if let Some(first_name) = self.priority.first() {
            if let Some((adapter, _)) = adapters.get(first_name) {
                return Ok(adapter.clone());
            }
        }

        Err(AdapterError::AllAdaptersFailed(
            "no adapters available".to_string(),
        ))
    }

    /// Get adapter health status by name
    pub async fn get_health(&self, name: &str) -> AdapterResult<AdapterHealth> {
        let adapters = self.adapters.read().await;
        if let Some((_, health)) = adapters.get(name) {
            let h = health.read().await;
            Ok(h.clone())
        } else {
            Err(AdapterError::AdapterNotFound(format!(
                "adapter not found: {}",
                name
            )))
        }
    }

    /// Update adapter health status
    pub async fn record_success(&self, name: &str, latency_ms: u32) -> AdapterResult<()> {
        let adapters = self.adapters.read().await;
        if let Some((_, health)) = adapters.get(name) {
            let mut h = health.write().await;
            h.record_success(latency_ms);
            Ok(())
        } else {
            Err(AdapterError::AdapterNotFound(format!(
                "adapter not found: {}",
                name
            )))
        }
    }

    /// Record adapter failure
    pub async fn record_failure(&self, name: &str) -> AdapterResult<()> {
        let adapters = self.adapters.read().await;
        if let Some((_, health)) = adapters.get(name) {
            let mut h = health.write().await;
            h.record_failure();
            Ok(())
        } else {
            Err(AdapterError::AdapterNotFound(format!(
                "adapter not found: {}",
                name
            )))
        }
    }

    /// List all registered adapters
    pub async fn list(&self) -> Vec<String> {
        let adapters = self.adapters.read().await;
        adapters.keys().cloned().collect()
    }

    /// List healthy adapters
    pub async fn list_healthy(&self) -> Vec<String> {
        let adapters = self.adapters.read().await;
        let mut healthy = vec![];
        for (name, (_, health)) in adapters.iter() {
            let h = health.read().await;
            if h.is_healthy {
                healthy.push(name.clone());
            }
        }
        healthy
    }

    /// Remove adapter by name
    pub async fn remove(&self, name: &str) -> AdapterResult<()> {
        let mut adapters = self.adapters.write().await;
        adapters.remove(name);
        Ok(())
    }

    /// Clear all adapters
    pub async fn clear(&self) -> AdapterResult<()> {
        let mut adapters = self.adapters.write().await;
        adapters.clear();
        Ok(())
    }

    /// Get adapter count
    pub async fn count(&self) -> usize {
        let adapters = self.adapters.read().await;
        adapters.len()
    }
}

/// Adapter factory for creating adapters with registry
pub struct AdapterFactory;

impl AdapterFactory {
    /// Create adapter instance from configuration
    pub async fn create(config: &AdapterConfig) -> AdapterResult<SharedAdapter> {
        match config.adapter_type.as_str() {
            "gemini" => {
                let adapter = GeminiAdapter::new(config.clone());
                Ok(Arc::new(tokio::sync::Mutex::new(adapter)))
            }
            "ollama" => {
                let adapter = OllamaAdapter::new(config.clone());
                Ok(Arc::new(tokio::sync::Mutex::new(adapter)))
            }
            _ => Err(AdapterError::AdapterNotFound(format!(
                "unknown adapter type: {}",
                config.adapter_type
            ))),
        }
    }

    /// Create registry with health checking
    pub async fn create_with_registry(config: &AdapterConfig) -> AdapterResult<Arc<AdapterRegistry>> {
        // Load adapter priority (hybrid: file or default)
        let priority = Self::load_priority().await.unwrap_or_default();

        // Create registry with priority
        let registry = AdapterRegistry::with_priority(priority.adapters);

        // Create and register adapters from priority list
        for adapter_name in registry.list().await {
            let mut adapter_config = config.clone();
            adapter_config.adapter_type = adapter_name.clone();

            if let Ok(adapter) = Self::create(&adapter_config).await {
                registry
                    .register_with_health(adapter_name, adapter)
                    .await?;
            }
        }

        // Spawn health checker background task
        let checker = HealthChecker::new(priority.check_interval_secs);
        // Note: In production, store the JoinHandle to manage task lifecycle
        let _health_check_handle = checker.spawn(
            Arc::new(tokio::sync::RwLock::new(
                registry
                    .adapters
                    .read()
                    .await
                    .iter()
                    .map(|(k, (a, h))| (k.clone(), (a.clone(), h.clone())))
                    .collect(),
            ))
        );

        Ok(registry)
    }

    async fn load_priority() -> AdapterResult<AdapterPriority> {
        // Try to load from adapters.toml in current directory
        if Path::new("adapters.toml").exists() {

                Err(_) => {} // Fall through to default

        Ok(AdapterPriority::default())

# M1.5.4: Adapter Registry & Discovery

## Overview

Production-grade adapter registry with health monitoring, configuration-driven selection, and automatic failover.

## Components

### AdapterHealth

Tracks health metrics for each adapter:
- **is_healthy**: Boolean flag (healthy if < 3 consecutive failures)
- **consecutive_failures**: Count since last success
- **success_count / error_count**: Lifetime stats
- **last_latency_ms**: Last recorded response time
- **success_rate()**: Calculate success percentage
- **average_latency_ms()**: Calculate avg latency
- **is_recently_checked()**: Check if within N seconds

### HealthChecker

Background task that periodically verifies adapter health:
- Spawns on factory creation
- Runs capability test on each adapter (send dummy request, measure latency)
- Updates health status in registry
- Configurable interval (default: 30 seconds)

### AdapterRegistry (Enhanced)

Core registry with health tracking and fallback logic:

#### Key Methods:
```rust
// Register adapter with health tracking
pub async fn register_with_health(&self, name: String, adapter: SharedAdapter) -> AdapterResult<()>

// Get best available adapter (respects priority, checks health)
pub async fn get_best_available(&self) -> AdapterResult<SharedAdapter>

// Get adapter health status
pub async fn get_health(&self, name: &str) -> AdapterResult<AdapterHealth>

// Record success/failure
pub async fn record_success(&self, name: &str, latency_ms: u32) -> AdapterResult<()>
pub async fn record_failure(&self, name: &str) -> AdapterResult<()>

// List healthy adapters
pub async fn list_healthy(&self) -> Vec<String>
```

#### Fallback Logic:
1. Iterate through priority list (e.g., gemini → ollama)
2. Return first adapter with `is_healthy = true`
3. If none healthy, return first in priority order (fallback)
4. If none exist, return `AllAdaptersFailed` error
