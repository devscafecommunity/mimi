## FlatBuffers Message Schema

### File: `proto/messages.fbs`

```flatbuffers
// Namespace
namespace MiMi.Messages;

// ============ Intent Message (Beatrice → Mimi) ============

table Intent {
  id: string (required);
  user_message: string (required);
  intent_type: IntentType;
  entities: [string];
  confidence: float;
  timestamp: uint64;
  metadata: Metadata;
}

enum IntentType : byte {
  Query = 0,
  Action = 1,
  SkillCreation = 2,
  Unknown = 3
}

table Metadata {
  source: string;
  user_id: string;
  session_id: string;
}

// ============ Task Message (Mimi → Adapters/Ryzu) ============

table Task {
  id: string (required);
  intent_id: string (required);
  task_type: TaskType;
  parameters: [Parameter];
  priority: Priority;
  deadline_ms: uint64;
  context: string;
}

enum TaskType : byte {
  LLMInference = 0,
  SkillExecution = 1,
  MemoryQuery = 2
}

enum Priority : byte {
  Low = 0,
  Medium = 1,
  High = 2
}

table Parameter {
  key: string (required);
  value: string (required);
  type: ParamType;
}

enum ParamType : byte {
  String = 0,
  Number = 1,
  Json = 2
}

// ============ Result Message (Adapter/Ryzu → Mimi) ============

table TaskResult {
  task_id: string (required);
  status: ResultStatus;
  output: string;
  error_message: string;
  execution_time_ms: uint64;
  tokens_used: uint32;
}

enum ResultStatus : byte {
  Success = 0,
  Failure = 1,
  Timeout = 2,
  Invalid = 3
}

// ============ Skill Message (Echidna → Odlaguna) ============

table SkillReview {
  skill_id: string (required);
  name: string (required);
  code: string (required);
  language: SkillLanguage;
  automability_score: float;
  repetitions: uint32;
}

enum SkillLanguage : byte {
  Rhai = 0,
  Wasm = 1
}

root_type Intent;
```

### Code: State Machine (src/core/state.rs)

```rust
use std::collections::HashMap;
use uuid::Uuid;

#[derive(Debug, Clone, PartialEq)]
pub enum MimiState {
    Idle,
    Processing(String),  // intent_id
    WaitingForResult(String),
    Error(String),
}

#[derive(Debug, Clone)]
pub struct TaskContext {
    pub id: String,
    pub intent_id: String,
    pub state: MimiState,
    pub created_at: std::time::SystemTime,
}

pub struct StateMachine {
    state: MimiState,
    contexts: HashMap<String, TaskContext>,
}

impl StateMachine {
    pub fn new() -> Self {
        Self {
            state: MimiState::Idle,
            contexts: HashMap::new(),
        }
    }

    pub fn process_intent(&mut self, intent_id: String) -> Result<String, String> {
        match &self.state {
            MimiState::Idle => {
                let task_id = Uuid::new_v4().to_string();
                self.state = MimiState::Processing(intent_id.clone());
                
                self.contexts.insert(task_id.clone(), TaskContext {
                    id: task_id.clone(),
                    intent_id,
                    state: MimiState::Processing(task_id.clone()),
                    created_at: std::time::SystemTime::now(),
                });
                
                Ok(task_id)
            }
            _ => Err("Mimi is busy".to_string()),
        }
    }

    pub fn mark_waiting(&mut self, task_id: String) {
        self.state = MimiState::WaitingForResult(task_id);
    }

    pub fn mark_complete(&mut self) {
        self.state = MimiState::Idle;
    }

    pub fn current_state(&self) -> &MimiState {
        &self.state
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_idle_to_processing() {
        let mut sm = StateMachine::new();
        let result = sm.process_intent("intent_123".to_string());
        assert!(result.is_ok());
        assert!(matches!(sm.current_state(), MimiState::Processing(_)));
    }

    #[test]
    fn test_cannot_process_while_busy() {
        let mut sm = StateMachine::new();
        sm.process_intent("intent_1".to_string()).unwrap();
        let result = sm.process_intent("intent_2".to_string());
        assert!(result.is_err());
    }
}
```

### Code: Bus Client (src/bus/client.rs)

```rust
use async_trait::async_trait;
use zenoh::prelude::r#async::*;
use std::sync::Arc;

#[async_trait]
pub trait BusClient: Send + Sync {
    async fn publish(&self, topic: &str, payload: &[u8]) -> Result<(), String>;
    async fn subscribe(&self, topic: &str) -> Result<BoxStream<Vec<u8>>, String>;
    async fn request_response(&self, topic: &str, payload: &[u8], timeout_ms: u64) -> Result<Vec<u8>, String>;
}

pub struct ZenohBusClient {
    session: Arc<Session>,
}

impl ZenohBusClient {
    pub async fn connect(endpoint: &str) -> Result<Self, String> {
        let config = zenoh::config::default();
        let session = zenoh::open(config)
            .res()
            .await
            .map_err(|e| format!("Failed to connect: {}", e))?;

        Ok(Self {
            session: Arc::new(session),
        })
    }
}

#[async_trait]
impl BusClient for ZenohBusClient {
    async fn publish(&self, topic: &str, payload: &[u8]) -> Result<(), String> {
        self.session
            .put(topic, payload.to_vec())
            .res()
            .await
            .map_err(|e| format!("Publish failed: {}", e))?;
        Ok(())
    }

    async fn subscribe(&self, topic: &str) -> Result<BoxStream<Vec<u8>>, String> {
        let subscriber = self.session
            .subscribe(topic)
            .res()
            .await
            .map_err(|e| format!("Subscribe failed: {}", e))?;

        let stream = futures::stream::unfold(subscriber, |sub| async move {
            match sub.recv_async().await {
                Ok(sample) => Some((sample.payload.to_vec(), sub)),
                Err(_) => None,
            }
        });

        Ok(Box::pin(stream))
    }

    async fn request_response(&self, topic: &str, payload: &[u8], timeout_ms: u64) -> Result<Vec<u8>, String> {
        let timeout = std::time::Duration::from_millis(timeout_ms);
        let response = self.session
            .get(topic)
            .with_value(payload.to_vec())
            .timeout(timeout)
            .res()
            .await
            .map_err(|e| format!("Request failed: {}", e))?;

        if let Ok(sample) = response {
            Ok(sample.payload.to_vec())
        } else {
            Err("No response received".to_string())
        }
    }
}
```

## Docker Setup

### File: `docker-compose.yml` (M1)

```yaml
version: '3.8'

services:
  # Zenoh Message Bus
  zenoh:
    image: zenoh/zenoh:latest
    container_name: mimi-zenoh
    ports:
      - "7447:7447"  # Zenoh default port
      - "7001:7001"  # HTTP management
    environment:
      - ZENOH_LOGLEVEL=info
    networks:
      - mimi-net
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:7001/info"]
      interval: 5s
      timeout: 2s
      retries: 3

  # Mimi Commander (will be built locally)
  mimi:
    build:
      context: .
      dockerfile: Dockerfile.mimi
    container_name: mimi-core
    depends_on:
      zenoh:
        condition: service_healthy
    environment:
      - BUS_ENDPOINT=zenoh://zenoh:7447
      - RUST_LOG=debug
    networks:
      - mimi-net
    ports:
      - "8080:8080"  # Mimi API (future)

networks:
  mimi-net:
    driver: bridge
```

## Integration Test Example

### File: `tests/e2e_flow.rs`

```rust
#[tokio::test]
async fn test_end_to_end_intent_processing() {
    // Setup
    let bus = ZenohBusClient::connect("127.0.0.1:7447").await.unwrap();
    let mut mimi = StateMachine::new();

    // Step 1: Create intent
    let intent_id = "intent_test_001";
    let result = mimi.process_intent(intent_id.to_string());
    assert!(result.is_ok());

    // Step 2: Publish intent via bus
    let payload = b"test_intent_data";
    bus.publish("intent/raw", payload).await.unwrap();

    // Step 3: Subscribe to result
    let mut results = bus.subscribe("task/result").await.unwrap();

    // Step 4: Wait for result (with timeout)
    let result = tokio::time::timeout(
        std::time::Duration::from_secs(5),
        results.next(),
    )
    .await;

    assert!(result.is_ok());
    mimi.mark_complete();
    assert!(matches!(mimi.current_state(), MimiState::Idle));
}
```
